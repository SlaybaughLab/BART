34 void AQBase<dim>::initialize_ref_bc_index ()
35 {
36  // Note: here we assume square domain and assume user either
37  // uses deal.II generated mesh or mesh from gmsh with proper
38  // boundary IDs setup: {0,1,2,3,4,5} for {xmin,xmax,ymin,ymax,zmin,zmax}
39  if (have_reflective_bc)
40  {
41  AssertThrow (dim>1,
42  ExcMessage("1D cases are not implemented for now."));
43  std::vector<Tensor<1, dim> > boundary_normal_vectors;
44  boundary_normal_vectors.resize (2*dim);
45  // All boundary normal vectors are assume to be parallel to axes
46  // Then, only one component in each normal vector is nonzero
47  if (dim==2)
48  {
49  boundary_normal_vectors[0][0] = -1.0;
50  boundary_normal_vectors[1][0] = 1.0;
51  boundary_normal_vectors[2][1] = -1.0;
52  boundary_normal_vectors[3][1] = 1.0;
53  }
54  if (dim==3)
55  {
56  boundary_normal_vectors[0][0] = -1.0;
57  boundary_normal_vectors[1][0] = 1.0;
58  boundary_normal_vectors[2][1] = -1.0;
59  boundary_normal_vectors[3][1] = 1.0;
60  boundary_normal_vectors[4][2] = -1.0;
61  boundary_normal_vectors[5][2] = 1.0;
62  }
63  for (unsigned int i=0; i<2*dim; ++i)
64  for (unsigned int i_dir=0; i_dir<n_dir; ++i_dir)
65  {
66  Tensor<1, dim> out_angle = omega_i[i_dir] - 2.0 * (boundary_normal_vectors[i] * omega_i[i_dir]) * boundary_normal_vectors[i];
67  //(omega_i[i_dir] *
68  // (1.0 - 2.0 * (boundary_normal_vectors[i] * omega_i[i_dir])));
69  for (unsigned int r_dir=0; r_dir<n_dir; ++r_dir)
70  {
71  Tensor<1, dim> d_dir = out_angle;
72  Tensor<1, dim> d_minus_dir = out_angle;
73  d_dir -= omega_i[r_dir];
74  d_minus_dir += omega_i[r_dir];
75  // Use caution about even parity using this std::map
76  if ((transport_model_name=="ep" &&
77  (d_dir.norm ()<1.0e-13 || d_minus_dir.norm ()<1.0e-13)) ||
78  (transport_model_name!="ep" && d_dir.norm ()<1.0e-13))
79  reflective_direction_index[std::make_pair (i, i_dir)] = r_dir;
80  }
81  }
82  }
